#include<bits/stdc++.h>
using namespace std;
const int span = 6000000;
vector<long long> prime;

void solution() {
    int n; cin >> n;
    vector<int> a(n);
    long long sum = 0;
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        sum += a[i];
    }
    sort(a.begin(), a.end());
    if(sum >= prime[n - 1]) {
        cout << 0 << endl;
        return;
    }

    for(int i = 0; i < n; i++) {
        sum -= a[i];
        if(sum >= prime[n - i - 2]) {
            cout << i + 1 << endl;
            return;
        }
    }
}
int main() {
    vector<bool> p(span, 1);
    for(int i = 2; i < span; i++) {
        if(!p[i]) continue;
        if(!prime.size()) {
            prime.push_back(i);
        }else
            prime.push_back(i + prime.back());
        for(int j = i; j < span; j+= i) {
            p[j] = 0;
        }
    }
    int t; cin >> t;
    while(t--) solution();
    return 0;
}
