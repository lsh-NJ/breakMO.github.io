#include<bits/stdc++.h>
using namespace std;
void solution() {
    int n; cin >> n;
    //分别记录二进制1相同位置1的数量和原数据
    vector<int> cnts(31, 0), a(n);

    for(int i = 0; i < n; i++) {
        cin >> a[i];
        for(int j = 0; j < 31; j++) {
            if((a[i] >> j) & 1) cnts[j]++;
        }
    }

    long long res = 0;
    //暴力遍历所有的数据取最大值
    for(int i = 0; i < n; i++) {
        int num = a[i];
        long long sum = 0;
        for(int j = 0; j < 31; j++) {
            bool tmp = (num >> j) & 1;
            if(tmp) sum += (n - cnts[j]) * pow(2, j);
            else sum += (cnts[j]) * pow(2, j);
        }
        res = max(res, sum);
    }

    cout << res << endl;
}

int main() {
    int t; cin >> t;
    while(t--) solution();
    return 0;
}
